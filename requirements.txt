# Quickr Python dependencies
# GTK3 / GObject / GdkPixbuf are system packages – install via your package manager:
#
#   Arch:   sudo pacman -S python-gobject gtk3 gdk-pixbuf2 librsvg
#   Debian: sudo apt install python3-gi python3-gi-cairo gir1.2-gtk-3.0 gir1.2-gdkpixbuf-2.0 librsvg2-common
#   Fedora: sudo dnf install python3-gobject gtk3 gdk-pixbuf2 librsvg2
#
# No PyPI packages are required beyond the system GTK bindings.
